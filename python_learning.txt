def album_name(a_name, b_name):
    full_details = f"{a_name} {b_name}"
    return full_details.title()

while True:
    print("\nGive me album details:")
    print("(enter q at any time to quit)")
    
    a_name = input("Album name: ")
    if a_name.lower() == "q":
        break

    b_name = input("Band name: ")
    if b_name.lower() == "q":
        break

    full_album_name = album_name(a_name, b_name)
    print(f"\nThe album is {full_album_name}")
